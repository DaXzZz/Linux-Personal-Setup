# 👤 USERNAME
[username]
style_user = "bg:#44475a fg:#f8f8f2"
format = '[ $user ]($style)'
show_always = true

# 📁 DIRECTORY
[directory]
style = "bg:#6272a4 fg:#f8f8f2"
format = "[ $path ]($style)"
truncation_length = 2
truncate_to_repo = true

# 🌿 GIT BRANCH
[git_branch]
style = "bg:#bd93f9 fg:#282a36"
format = '[  $branch$git_status ]($style)'

# 🔄 GIT STATUS 
[git_status]
style = "bg:#bd93f9 fg:#282a36"
format = '([$all_status$ahead_behind]($style))'
conflicted = "C"  # Conflict
staged = "+"      # Staged
modified = "*"    # Modified
deleted = "-"     # Deleted
renamed = "R"     # Renamed
untracked = "?"   # Untracked
stashed = "S"     # Stashed

# ⌛ CMD DURATION
[cmd_duration]
min_time = 500
style = "bg:#ff79c6 fg:#282a36"
format = '[ ⏱ $duration ]($style)'

# 🧾 STATUS (Exit Code)
[status]
#style = "bg:#ff5555 fg:#282a36"

# ⚙️ JOBS
[jobs]
symbol = "⚙️ "

#  SHELL
[shell]
style = "dimmed cyan"

# 📦 PACKAGE VERSION
[package]

# 🐳 DOCKER CONTEXT
[docker_context]

#  NODEJS
[nodejs]
format = "[  $version](bold green) "

# 🦀 RUST

#  PYTHON
[python]

[character]
